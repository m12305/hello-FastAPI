## 2. 基于角色的访问控制（RBAC）

RBAC（Role-Based Access Control）是最常用的权限模型：

```
┌──────────────────────────────────┐
│           Super Admin             │  ← 一切权限
│  ┌──────────────────────────────┐│
│  │           Admin               ││  ← 管理用户、管理内容
│  │  ┌──────────────────────────┐││
│  │  │         Editor            │││  ← 编辑内容
│  │  │  ┌──────────────────────┐│││
│  │  │  │        User           ││││  ← 只能操作自己的数据
│  │  │  └──────────────────────┘│││
│  │  └──────────────────────────┘││
│  └──────────────────────────────┘│
└──────────────────────────────────┘
```

### 模型设计

```python
from enum import Enum
from sqlalchemy import String, Enum as SAEnum
from sqlalchemy.orm import Mapped, mapped_column


class UserRole(str, Enum):
    USER = "user"
    EDITOR = "editor"
    ADMIN = "admin"
    SUPER_ADMIN = "super_admin"


class User(Base):
    __tablename__ = "users"

    id: Mapped[int] = mapped_column(primary_key=True)
    username: Mapped[str] = mapped_column(String(50), unique=True)
    role: Mapped[UserRole] = mapped_column(
        SAEnum(UserRole),  # ← 用 SQLAlchemy Enum 存角色
        default=UserRole.USER,
    )
    # ... 其他字段 ...
```

## 3. 权限依赖层级设计

### 3.1 基本角色依赖

```python
from fastapi import Depends, HTTPException, status

# ═══════ 层 1: 认证（需要登录） ═══════
def get_current_user(token: str = Depends(oauth2_scheme), db = Depends(get_db)) -> User:
    """基础认证——只要登录了就能过"""
    # ... JWT 验证逻辑 ...
    return user


# ═══════ 层 2: 角色检查（需要特定角色） ═══════
class RoleChecker:
    """
    角色检查器——工厂类，传入角色，返回该角色的校验依赖

    用法:
        require_admin = RoleChecker(["admin"])
        @app.get("/admin")
        def admin_panel(user = Depends(require_admin)):
            ...
    """
    def __init__(self, allowed_roles: list[str]):
        self.allowed_roles = allowed_roles

    def __call__(self, current_user: User = Depends(get_current_user)):
        if current_user.role not in self.allowed_roles:
            raise HTTPException(
                status_code=status.HTTP_403_FORBIDDEN,
                detail=f"权限不足！需要 {self.allowed_roles} 角色，当前角色: {current_user.role}",
            )
        return current_user


# ═══════ 预定义的角色检查依赖 ═══════
require_admin = RoleChecker(["admin", "super_admin"])
require_editor = RoleChecker(["editor", "admin", "super_admin"])
require_super_admin = RoleChecker(["super_admin"])
```

### 3.2 使用角色依赖

```python
# 任何登录用户都能访问
@app.get("/me")
def read_me(current_user: User = Depends(get_current_user)):
    return {"username": current_user.username}


# 只有编辑者和更高角色能创建内容
@app.post("/posts/")
def create_post(
    post_in: PostCreate,
    current_user: User = Depends(require_editor),
):
    return {"message": f"{current_user.username} 创建了文章"}


# 只有管理员能看统计数据
@app.get("/admin/stats")
def admin_stats(current_user: User = Depends(require_admin)):
    return {"total_users": 1250, "total_posts": 890}


# 只有超级管理员能删除用户
@app.delete("/admin/users/{user_id}")
def delete_user(user_id: int, current_user: User = Depends(require_super_admin)):
    return {"message": f"用户 {user_id} 已删除"}
```

## 4. 数据所有权检查

角色权限还不够——用户只能操作**自己的**数据：

```python
def get_own_post(
    post_id: int,
    current_user: User = Depends(get_current_user),
    db: Session = Depends(get_db),
) -> Post:
    """
    获取文章——且必须属于当前用户。

    管理员可以跳过此检查（用 require_admin）。
    """
    post = db.get(Post, post_id)
    if not post:
        raise HTTPException(status_code=404, detail="文章不存在")

    # 检查所有权
    if post.user_id != current_user.id:
        raise HTTPException(
            status_code=status.HTTP_403_FORBIDDEN,
            detail="你只能操作自己的文章",
        )
    return post


# ── 使用 ──
@app.patch("/posts/{post_id}")
def update_own_post(
    post: Post = Depends(get_own_post),  # ← 所有权 + 认证一步到位
    post_in: PostUpdate = None,
):
    """更新自己的文章"""
    # post 已经通过所有权验证
    post.title = post_in.title or post.title
    return post


@app.patch("/admin/posts/{post_id}")
def admin_update_any_post(
    post_id: int,
    post_in: PostUpdate,
    admin: User = Depends(require_admin),  # ← 管理员可以修改任何文章
    db: Session = Depends(get_db),
):
    """管理员更新任何文章"""
    post = db.get(Post, post_id)
    if not post:
        raise HTTPException(status_code=404, detail="文章不存在")
    post.title = post_in.title or post.title
    return post
```

## 5. 权限装饰器模式

如果嫌 `Depends(require_admin)` 不够直观，可以自定义装饰器：

```python
from functools import wraps


def require_role(role: str):
    """
    权限装饰器——但注意：FastAPI 中装饰器不能替换 Depends

    更推荐使用 Depends() 依赖注入方式。
    这个示例仅用于展示"概念"，实际请用 RoleChecker。
    """
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            current_user = kwargs.get("current_user")
            if not current_user or current_user.role != role:
                raise HTTPException(status_code=403, detail="权限不足")
            return await func(*args, **kwargs)
        return wrapper
    return decorator
```

> **注意：** FastAPI 的依赖注入（`Depends()`）是实现权限控制的最佳方式。Python 装饰器在 FastAPI 中不如依赖注入灵活，因为 FastAPI 在函数调用前就需要解析参数。

## 6. API Key 认证（机器间通信）

对于机器间（Server-to-Server）通信，JWT 太重了——API Key 更合适。

### 6.1 API Key 认证实现

```python
"""api_key_auth.py — API Key 认证"""
from fastapi import FastAPI, Depends, HTTPException, Security
from fastapi.security import APIKeyHeader
from sqlalchemy.orm import Session

# 从请求头 X-API-Key 读取
api_key_header = APIKeyHeader(name="X-API-Key", auto_error=False)

# 模拟 API Key 存储（实际存数据库）
VALID_API_KEYS = {
    "key-service-a-abc123": {"name": "Service A", "permissions": ["read:users"]},
    "key-service-b-xyz789": {"name": "Service B", "permissions": ["read:users", "write:orders"]},
}


def get_api_client(
    api_key: str | None = Security(api_key_header),
):
    """验证 API Key——用于服务间调用"""
    if not api_key:
        raise HTTPException(status_code=401, detail="缺少 X-API-Key 请求头")

    client_info = VALID_API_KEYS.get(api_key)
    if not client_info:
        raise HTTPException(status_code=403, detail="无效的 API Key")

    return client_info


# ── 使用 ──

@app.get("/api/external/users")
def external_users_api(client: dict = Depends(get_api_client)):
    """供外部服务调用的 API——不需要 JWT，用 API Key"""
    if "read:users" not in client["permissions"]:
        raise HTTPException(status_code=403, detail="该 API Key 无权访问用户数据")
    return {"source": client["name"], "users": [...]}
```

### 6.2 API Key vs JWT

| | API Key | JWT |
|------|---------|-----|
| **适用场景** | 机器间通信、第三方集成 | 用户认证 |
| **结构** | 随机字符串 | 三段式编码 |
| **是否自包含** | 否（需要查数据库） | 是（Token 内含用户信息） |
| **过期机制** | 手动管理 | Token 自带 exp 字段 |
| **权限粒度** | 粗粒度（服务级） | 细粒度（用户级） |

## 7. 多认证方式共存

一个真实的应用可能同时支持多种认证方式：

```python
"""multi_auth.py — 用户 JWT + 服务 API Key 共存"""
from fastapi import Depends, HTTPException, Security
from fastapi.security import OAuth2PasswordBearer, APIKeyHeader

oauth2_scheme = OAuth2PasswordBearer(tokenUrl="/login", auto_error=False)
api_key_header = APIKeyHeader(name="X-API-Key", auto_error=False)


async def get_current_identity(
    # 两个认证源都设为 auto_error=False，允许多种方式
    token: str | None = Security(oauth2_scheme),
    api_key: str | None = Security(api_key_header),
    db=Depends(get_db),
) -> dict:
    """
    统一认证入口——支持 JWT 或 API Key。

    返回统一格式的 identity:
      {"type": "user", "user": User(...)}
      {"type": "service", "name": "Service A"}
    """
    # 优先检查 JWT（用户认证）
    if token:
        try:
            payload = decode_access_token(token)
            user = db.get(User, int(payload["sub"]))
            if user:
                return {"type": "user", "user": user}
        except Exception:
            pass  # JWT 失败了，再试 API Key

    # 再检查 API Key（服务认证）
    if api_key:
        client = VALID_API_KEYS.get(api_key)
        if client:
            return {"type": "service", "name": client["name"]}

    # 都失败了
    raise HTTPException(
        status_code=401,
        detail="请提供有效的 Bearer Token 或 X-API-Key",
        headers={"WWW-Authenticate": "Bearer"},
    )


# ── 使用 ──
@app.get("/hybrid-endpoint")
def hybrid_endpoint(identity: dict = Depends(get_current_identity)):
    """用户和服务都能访问的端点"""
    if identity["type"] == "user":
        return {"message": f"欢迎用户 {identity['user'].username}"}
    else:
        return {"message": f"来自服务 {identity['name']} 的请求"}
```

## 8. 细粒度权限：Scope（作用域）

OAuth2 的 Scope 提供了比角色更细粒度的权限控制：

```python
# 权限定义
SCOPES = {
    "users:read": "查看用户信息",
    "users:write": "创建/修改用户",
    "users:delete": "删除用户",
    "posts:read": "查看文章",
    "posts:write": "创建/修改文章",
    "admin:all": "管理员全权限",
}

# 角色 → Scope 映射
ROLE_SCOPES = {
    "user": ["users:read", "posts:read"],
    "editor": ["users:read", "posts:read", "posts:write"],
    "admin": ["users:read", "users:write", "posts:read", "posts:write", "admin:all"],
}


def require_scope(required_scope: str):
    """Scope 权限检查依赖——比角色更细粒度"""
    def scope_checker(current_user: User = Depends(get_current_user)):
        user_scopes = ROLE_SCOPES.get(current_user.role, [])
        if required_scope not in user_scopes:
            raise HTTPException(
                status_code=403,
                detail=f"需要权限: {required_scope}，你当前角色: {current_user.role}",
            )
        return current_user
    return scope_checker


# ── 使用 ──
@app.delete("/users/{user_id}")
def delete_user(
    user_id: int,
    current_user: User = Depends(require_scope("users:delete")),
):
    """删除用户——需要 users:delete 权限"""
    ...
```
