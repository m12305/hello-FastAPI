## 1. 索引设计与查询优化

### 为什么索引重要？

```
没有索引：全表扫描
  SELECT * FROM users WHERE email = 'zhang@example.com';
  数据库行为：从头到尾一行一行检查 → 100 万行 = 慢

有索引：B-Tree 查找
  SELECT * FROM users WHERE email = 'zhang@example.com';
  数据库行为：跳转几次就找到 → O(log n) ≈ 毫秒级
```

### 在 SQLAlchemy 中创建索引

```python
from sqlalchemy import Index

class User(Base):
    __tablename__ = "users"

    id: Mapped[int] = mapped_column(primary_key=True)
    username: Mapped[str] = mapped_column(String(50), unique=True, index=True)  # ← 方式 1
    email: Mapped[str] = mapped_column(String(100), index=True)

    # 方式 2：复合索引（跨多列的查询优化）
    __table_args__ = (
        Index("ix_users_email_active", "email", "is_active"),   # email + is_active 组合查询
        Index("ix_users_created", "created_at", postgresql_using="brin"),  # 时序数据用 BRIN
    )
```

### 索引策略速查

| 查询模式 | 建议索引 |
|----------|---------|
| `WHERE email = ?` | 单列索引 `email` |
| `WHERE email = ? AND is_active = ?` | 复合索引 `(email, is_active)` |
| `WHERE name LIKE '张%'` | 单列索引 `name`（前缀匹配可用索引） |
| `WHERE name LIKE '%三'` | ❌ 索引无效！以通配符开头 |
| `ORDER BY created_at DESC` | 单列索引 `created_at` |
| `JOIN ... ON user_id = ?` | 索引外键列 |

### 用 EXPLAIN 分析查询

```python
from sqlalchemy import text

# 在同步 session 中
result = session.execute(
    text("EXPLAIN QUERY PLAN SELECT * FROM users WHERE email = :email"),
    {"email": "zhang@example.com"}
)
for row in result:
    print(row[0])
# 输出示例：
# SEARCH users USING INDEX ix_users_email (email=?)
# 看到 USING INDEX 说明索引生效 ✅
# 看到 SCAN 说明全表扫描 ❌
```

## 2. 事务管理

### 事务是什么？

```
事务 = 一组要么全部成功、要么全部回滚的操作

转账示例：
  张三账户 -1000 元 ──┐
                      ├── 必须同时成功或同时失败
  李四账户 +1000 元 ──┘
```

### 在 FastAPI 中管理事务

```python
from fastapi import Depends, HTTPException
from sqlalchemy.orm import Session
from database import get_db


def transfer_money(
    from_user_id: int,
    to_user_id: int,
    amount: float,
    db: Session = Depends(get_db),
):
    """转账——事务要么全部成功，要么全部回滚"""
    from_user = db.get(User, from_user_id)
    to_user = db.get(User, to_user_id)

    if not from_user or not to_user:
        raise HTTPException(404, "用户不存在")

    if from_user.balance < amount:
        raise HTTPException(400, "余额不足")

    try:
        from_user.balance -= amount   # 扣钱
        to_user.balance += amount     # 加钱

        # 记录转账日志
        log = TransferLog(
            from_user_id=from_user_id,
            to_user_id=to_user_id,
            amount=amount,
        )
        db.add(log)

        db.commit()  # ← 三个操作一起提交
        return {"status": "success"}

    except Exception:
        db.rollback()  # ← 任何一个失败，全部撤销
        raise HTTPException(500, "转账失败，已回滚")
```

### 嵌套事务（Savepoint）

```python
def complex_operation(db: Session):
    """主事务中创建保存点——部分失败不影响整体"""
    db.begin_nested()  # 创建 savepoint
    try:
        # 尝试操作 A（允许失败）
        db.execute(some_risky_sql)
    except:
        db.rollback()  # 只回滚 savepoint，不回滚整个事务

    # 继续操作 B（不会受 A 失败影响）
    db.add(something_else)
    db.commit()  # B 仍然成功提交
```

### 事务隔离级别

```python
engine = create_engine(
    DATABASE_URL,
    isolation_level="READ COMMITTED",  # PostgreSQL 默认
    # 其他选项:
    #   "SERIALIZABLE"      — 最严格，最慢（"转账"这类场景）

    #   "READ COMMITTED"    — 平衡（大多数 Web 应用）
    #   "READ UNCOMMITTED"  — 最宽松，最快（仅日志/分析）
```

## 3. N+1 查询问题——彻底解决

### 问题再现

```python
# ❌ N+1 查询：查 100 篇文章，发 101 次 SQL
posts = db.scalars(select(Post)).all()   # 1 次查询：拿到 100 篇文章
for post in posts:
    print(post.author.username)          # 100 次查询：每篇查一次作者！
```

### 解决方案全集

```python
from sqlalchemy.orm import joinedload, selectinload, subqueryload


# 方案 1：joinedload —— LEFT JOIN，一次 SQL 全拿回来
# 适用：多对一 / 一对一
stmt = select(Post).options(joinedload(Post.author))
posts = db.scalars(stmt).unique().all()
# SQL: SELECT posts.*, users.* FROM posts LEFT JOIN users ON ...


# 方案 2：selectinload —— 第二条 SQL 用 IN 查关联表
# 适用：一对多 / 多对多（避免 JOIN 产生的笛卡尔积）
stmt = select(User).options(selectinload(User.posts))
users = db.scalars(stmt).unique().all()
# SQL 1: SELECT * FROM users
# SQL 2: SELECT * FROM posts WHERE user_id IN (1, 2, 3, ...)


# 方案 3：两级关联加载
stmt = (
    select(Post)
    .options(
        joinedload(Post.author),               # 加载作者
        selectinload(Post.comments)            # 加载评论（一对多）
        .joinedload(Comment.author),           # 加载评论的作者
    )
)
posts = db.scalars(stmt).unique().all()
```

### 选择策略的决策树

```
查询关联对象？
├── 一对一 / 多对一 → joinedload()
├── 一对多 / 多对多 → selectinload()
├── 多层嵌套关联 → joinedload() 用于 x-to-one + selectinload() 用于 x-to-many
└── 不确定要不要加载 → 默认 lazy，按需改 eager
```

## 4. 分页——通用实现

### Offset 分页（最常见）

```python
from typing import TypeVar, Generic
from pydantic import BaseModel

T = TypeVar("T")


class Page(BaseModel, Generic[T]):
    """通用分页响应模型"""
    items: list[T]
    total: int
    page: int
    page_size: int
    total_pages: int

    @property
    def has_next(self) -> bool:
        return self.page < self.total_pages

    @property
    def has_prev(self) -> bool:
        return self.page > 1


def paginate(
    db: Session,
    stmt,
    page: int = 1,
    page_size: int = 20,
) -> dict:
    """通用 offset 分页函数

    Args:
        db: 数据库 session
        stmt: SQLAlchemy select 语句（不需要加 limit/offset）
        page: 页码（从 1 开始）
        page_size: 每页数量

    Returns:
        {"items": [...], "total": N, "page": page, "page_size": page_size, "total_pages": M}
    """
    # 先查总数
    count_stmt = stmt.with_only_columns(func.count()).order_by(None)
    total = db.scalar(count_stmt)

    # 分页查询
    offset = (page - 1) * page_size
    items = db.scalars(stmt.offset(offset).limit(page_size)).all()

    return {
        "items": items,
        "total": total,
        "page": page,
        "page_size": page_size,
        "total_pages": (total + page_size - 1) // page_size,  # 向上取整
    }
```

```python
# 在端点中使用
from sqlalchemy import func

@app.get("/posts/", response_model=Page[PostResponse])
def read_posts(
    page: int = 1,
    page_size: int = 20,
    db: Session = Depends(get_db),
):
    stmt = select(Post).where(Post.published == True).order_by(Post.created_at.desc())
    return paginate(db, stmt, page=page, page_size=page_size)
```

### Cursor 分页（大数据量场景）

```python
def paginate_cursor(
    db: Session,
    stmt,
    cursor_field,       # 用于分页的字段（通常是有索引的时间/ID 列）
    cursor_value=None,  # 上一页最后一项的 cursor 值
    page_size: int = 20,
):
    """Cursor 分页——用 WHERE > cursor 代替 OFFSET，性能更好"""
    if cursor_value is not None:
        stmt = stmt.where(cursor_field > cursor_value)

    items = db.scalars(
        stmt.order_by(cursor_field.asc()).limit(page_size + 1)  # 多查一条判断是否有下一页
    ).all()

    has_next = len(items) > page_size
    if has_next:
        items = items[:page_size]

    return {
        "items": items,
        "has_next": has_next,
        "next_cursor": str(getattr(items[-1], cursor_field.key)) if items else None,
    }
```

## 5. 软删除（Soft Delete）

### 为什么需要软删除？

```
硬删除（DELETE）：
  ✅ 省空间
  ❌ 数据不可恢复
  ❌ 关联数据可能出问题
  ❌ 无法保留删除审计记录

软删除（标记 is_deleted）：
  ✅ 数据可恢复（"回收站"功能）
  ✅ 保留审计记录
  ✅ 关联数据不受影响
  ❌ 需要额外的存储空间
  ❌ 所有查询都要加 WHERE is_deleted = false
```

### 实现：基于 Mixin 的软删除

```python
from datetime import datetime
from sqlalchemy import Boolean, DateTime, func
from sqlalchemy.orm import Mapped, mapped_column


class SoftDeleteMixin:
    """软删除 Mixin——嵌入任何需要软删除的模型"""
    is_deleted: Mapped[bool] = mapped_column(
        Boolean, default=False, index=True
    )
    deleted_at: Mapped[datetime | None] = mapped_column(
        DateTime, default=None
    )

    def soft_delete(self):
        """标记为已删除"""
        self.is_deleted = True
        self.deleted_at = func.now()


class User(Base, SoftDeleteMixin):
    __tablename__ = "users"
    # ... 其他字段 ...


class Post(Base, SoftDeleteMixin):
    __tablename__ = "posts"
    # ... 其他字段 ...
```

### CRUD 适配

```python
def get_active_users(db: Session, skip: int = 0, limit: int = 100) -> list[User]:
    """查询时自动过滤已删除的记录"""
    stmt = (
        select(User)
        .where(User.is_deleted == False)  # ← 关键过滤
        .offset(skip)
        .limit(limit)
    )
    return list(db.scalars(stmt).all())


def soft_delete_user(db: Session, user_id: int) -> bool:
    """软删除用户"""
    user = db.get(User, user_id)
    if not user or user.is_deleted:
        return False
    user.soft_delete()
    db.commit()
    return True


def restore_user(db: Session, user_id: int) -> bool:
    """恢复已删除的用户（回收站）"""
    user = db.get(User, user_id)
    if not user or not user.is_deleted:
        return False
    user.is_deleted = False
    user.deleted_at = None
    db.commit()
    return True
```

## 6. Repository 模式

### 为什么需要 Repository？

当前我们的 `crud.py` 已经提取了数据库操作，但还存在问题：

```python
# crud.py 中的重复代码模式
def get_user(db, id): ...     # User 的 get
def get_post(db, id): ...     # Post 的 get —— 几乎一模一样！
def get_comment(db, id): ...  # Comment 的 get —— 又是重复！
```

Repository 模式将**通用的数据库操作抽象到基类中**。

### 通用 Repository 基类

```python
from typing import TypeVar, Generic
from sqlalchemy.orm import Session
from sqlalchemy import select, func

T = TypeVar("T")


class BaseRepository(Generic[T]):
    """通用 Repository 基类"""

    def __init__(self, model: type[T], db: Session):
        self.model = model
        self.db = db

    # ─── Read ───
    def get(self, id: int) -> T | None:
        return self.db.get(self.model, id)

    def get_all(self, skip: int = 0, limit: int = 100) -> list[T]:
        stmt = select(self.model).offset(skip).limit(limit)
        return list(self.db.scalars(stmt).all())

    def find_by(self, **kwargs) -> list[T]:
        """按条件查询，如 find_by(is_active=True, role="admin")"""
        stmt = select(self.model).filter_by(**kwargs)
        return list(self.db.scalars(stmt).all())

    def find_one_by(self, **kwargs) -> T | None:
        stmt = select(self.model).filter_by(**kwargs)
        return self.db.scalars(stmt).first()

    def count(self) -> int:
        stmt = select(func.count()).select_from(self.model)
        return self.db.scalar(stmt)

    # ─── Create ───
    def create(self, **kwargs) -> T:
        instance = self.model(**kwargs)
        self.db.add(instance)
        self.db.commit()
        self.db.refresh(instance)
        return instance

    def bulk_create(self, items: list[dict]) -> list[T]:
        instances = [self.model(**item) for item in items]
        self.db.add_all(instances)
        self.db.commit()
        return instances

    # ─── Update ───
    def update(self, instance: T, **kwargs) -> T:
        for key, value in kwargs.items():
            setattr(instance, key, value)
        self.db.commit()
        self.db.refresh(instance)
        return instance

    # ─── Delete ───
    def delete(self, instance: T) -> None:
        self.db.delete(instance)
        self.db.commit()

    def delete_by_id(self, id: int) -> bool:
        instance = self.get(id)
        if instance:
            self.delete(instance)
            return True
        return False
```

### 使用 Repository

```python
"""routers/users.py — 使用 Repository 的端点"""
from fastapi import APIRouter, Depends
from repositories import BaseRepository
from models import User
from schemas import UserCreate, UserResponse

router = APIRouter(prefix="/users", tags=["用户管理"])


def get_user_repo(db=Depends(get_db)) -> BaseRepository[User]:
    """依赖注入：获取 User Repository"""
    return BaseRepository(User, db)


@router.post("/", response_model=UserResponse, status_code=201)
def create_user(
    user_in: UserCreate,
    repo: BaseRepository[User] = Depends(get_user_repo),
):
    existing = repo.find_one_by(email=user_in.email)
    if existing:
        raise HTTPException(400, "邮箱已存在")
    return repo.create(
        username=user_in.username,
        email=user_in.email,
        hashed_password="hashed_" + user_in.password,
    )


@router.get("/", response_model=list[UserResponse])
def read_users(
    skip: int = 0,
    limit: int = 100,
    repo: BaseRepository[User] = Depends(get_user_repo),
):
    return repo.get_all(skip=skip, limit=limit)


@router.get("/{user_id}", response_model=UserResponse)
def read_user(
    user_id: int,
    repo: BaseRepository[User] = Depends(get_user_repo),
):
    user = repo.get(user_id)
    if not user:
        raise HTTPException(404, "用户不存在")
    return user
```

```python
        if user:
            return self.update(user, is_active=False)
        return None
```

```python
"""3.5-advanced-demo.py — 高级话题综合演示"""
from sqlalchemy import create_engine, String, Integer, Boolean, DateTime, func, select, Index
from sqlalchemy.orm import DeclarativeBase, Mapped, mapped_column, Session, sessionmaker
from datetime import datetime

engine = create_engine("sqlite:///./advanced.db", echo=False)
SessionLocal = sessionmaker(bind=engine)


class Base(DeclarativeBase):
    pass


# ════════ 软删除 Mixin ════════
class SoftDeleteMixin:
    is_deleted: Mapped[bool] = mapped_column(Boolean, default=False, index=True)
    deleted_at: Mapped[datetime | None] = mapped_column(DateTime, default=None)

    def soft_delete(self):
        self.is_deleted = True
        self.deleted_at = datetime.utcnow()


# ════════ 模型 ════════
class User(Base, SoftDeleteMixin):
    __tablename__ = "users"
    __table_args__ = (
        Index("ix_email_active", "email", "is_active"),
    )

    id: Mapped[int] = mapped_column(primary_key=True)
    username: Mapped[str] = mapped_column(String(50), unique=True, index=True)
    email: Mapped[str] = mapped_column(String(100), unique=True, index=True)
    is_active: Mapped[bool] = mapped_column(default=True)
    created_at: Mapped[datetime] = mapped_column(DateTime, server_default=func.now())


# ════════ Repository ════════
class BaseRepository:
    def __init__(self, model, db: Session):
        self.model = model
        self.db = db

    def get(self, id: int):
        return self.db.get(self.model, id)

    def get_all(self, skip=0, limit=100):
        stmt = select(self.model).offset(skip).limit(limit)
        return list(self.db.scalars(stmt).all())

    def create(self, **kwargs):
        instance = self.model(**kwargs)
        self.db.add(instance)
        self.db.commit()
        self.db.refresh(instance)
        return instance

    def update(self, instance, **kwargs):
        for k, v in kwargs.items():
            setattr(instance, k, v)
        self.db.commit()
        self.db.refresh(instance)
        return instance

    def delete(self, instance):
        self.db.delete(instance)
        self.db.commit()


# ════════ 分页工具 ════════
def paginate(db: Session, stmt, page=1, page_size=20):
    count_stmt = stmt.with_only_columns(func.count()).order_by(None)
    total = db.scalar(count_stmt)
    items = db.scalars(
        stmt.offset((page - 1) * page_size).limit(page_size)
    ).all()
    return {
        "items": items,
        "total": total,
        "page": page,
        "page_size": page_size,
        "total_pages": (total + page_size - 1) // page_size,
    }


# ════════ 演示 ════════
def demo():
    Base.metadata.create_all(bind=engine)
    db = SessionLocal()
    repo = BaseRepository(User, db)

    try:
        # 创建
        user = repo.create(username="张三", email="zhang@example.com")
        print(f"创建: {user.username}")

        # 软删除
        user.soft_delete()
        db.commit()
        print(f"软删除: is_deleted={user.is_deleted}")

        # 分页查询
        stmt = select(User).where(User.is_active == True)
        result = paginate(db, stmt, page=1, page_size=10)
        print(f"分页: {result['total']} 用户, {result['total_pages']} 页")

    finally:
        db.close()


if __name__ == "__main__":
    demo()
```
